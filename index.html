        private void checkEmployeeForExistence(EmployeeDto dto) {
            Long id = dto.getId();
            try {
                this.findById(id);
            } catch (Exception e) {
                String message = MessageFormat.format("Such employee id={0} not found", id);
                LOGGER.error(message);
                throw new NoSuchElementException(message);
            }
        }
